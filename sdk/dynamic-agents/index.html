    Dynamic Agents

    Create and modify agents at runtime based on context, user needs, or system conditions.

    Overview

    Dynamic agents allow you to: - Create agents programmatically during execution - Modify agent capabilities based on context - Scale agent instances based on load - Implement adaptive agent behaviors

    Dynamic Agent Creation

    Runtime Agent Creation

    from robutler.agents import BaseAgent
from robutler.server import app

async def create_specialized_agent(domain: str, user_id: str):
    """Create agent tailored to specific domain and user"""

    # Determine skills based on domain
    skills = {}

    if domain == "finance":
        from robutler.agents.skills import DatabaseSkill, PaymentSkill
        skills.update({
            "database": DatabaseSkill({"connection": "finance_db"}),
            "payment": PaymentSkill()
        })
    elif domain == "healthcare":
        from robutler.agents.skills import GuardrailsSkill
        skills.update({
            "guardrails": GuardrailsSkill({"strict_mode": True})
        })

    # Create personalized agent
    agent = BaseAgent(
        name=f"{domain}-expert-{user_id}",
        instructions=f"You are a {domain} expert assistant specialized for user {user_id}",
        model="openai/gpt-4o",
        skills=skills
    )

    # Register with server (if using the server package)
    app.register_agent(agent)

    return agent

    class AgentFactory:
    """Factory for creating different types of agents"""

    @staticmethod
    def create_customer_service_agent(tier: str = "basic") -> BaseAgent:
        """Create customer service agent based on tier"""

        from robutler.agents.skills import (
            ShortTermMemorySkill, 
            DiscoverySkill,
            PaymentSkill
        )

        skills = {
            "memory": ShortTermMemorySkill({"max_messages": 50})
        }

        if tier in ["premium", "enterprise"]:
            skills.update({
                "discovery": DiscoverySkill(),
                "payment": PaymentSkill()
            })

        instructions = "You are a customer service agent."
        if tier == "enterprise":
            instructions += " You have access to advanced tools and can escalate to specialists."

        return BaseAgent(
            name=f"customer-service-{tier}",
            instructions=instructions,
            model="openai/gpt-4o",
            skills=skills
        )

    @staticmethod
    def create_data_analyst(datasets: List[str]) -> BaseAgent:
        """Create data analyst with access to specific datasets"""

        from robutler.agents.skills import DatabaseSkill

        # Configure database access for specific datasets
        db_config = {
            "allowed_tables": datasets,
            "read_only": True
        }

        return BaseAgent(
            name=f"analyst-{'-'.join(datasets[:2])}",
            instructions=f"You are a data analyst with access to: {', '.join(datasets)}",
            model="anthropic/claude-3-sonnet",
            skills={
                "database": DatabaseSkill(db_config)
            }
        )

    User-Specific Customization

    from robutler.agents.skills import Skill
from robutler.agents.skills.decorators import hook

class AdaptiveAgentSkill(Skill):
    """Skill that adapts agent behavior to user context"""

    @hook("on_connection", priority=1)
    async def adapt_to_user(self, context):
        """Adapt agent based on user profile"""

        user_id = context.peer_user_id
        user_profile = await self.get_user_profile(user_id)

        # Adapt based on user preferences
        if user_profile.get("expertise_level") == "expert":
            # Add advanced tools for expert users
            self.register_tool(self.advanced_analysis)

            # Update prompts for technical audience
            context["user_prompts"] = [
                "Use technical terminology when appropriate.",
                "Provide detailed explanations with implementation details."
            ]

        elif user_profile.get("expertise_level") == "beginner":
            # Add helper tools for beginners
            self.register_tool(self.explain_concepts)

            # Update prompts for non-technical audience
            context["user_prompts"] = [
                "Use simple, non-technical language.",
                "Provide step-by-step explanations.",
                "Always ask if clarification is needed."
            ]

        # Adapt based on user's industry
        industry = user_profile.get("industry")
        if industry == "healthcare":
            self.register_tool(self.medical_terminology_tool)
        elif industry == "finance":
            self.register_tool(self.financial_calculations_tool)

        return context

    @tool
    def advanced_analysis(self, data: str) -> str:
        """Advanced analysis tool for expert users"""
        return "Detailed technical analysis..."

    @tool 
    def explain_concepts(self, concept: str) -> str:
        """Explain concepts in simple terms"""
        return f"Simple explanation of {concept}..."

    Load-Based Scaling

    Agent Pool Management

    import asyncio
from typing import Dict, List
from dataclasses import dataclass

@dataclass
class AgentPool:
    name: str
    template: BaseAgent
    instances: List[BaseAgent]
    max_instances: int = 10
    min_instances: int = 1

class DynamicAgentManager:
    """Manage dynamic agent scaling and load balancing"""

    def __init__(self):
        self.pools: Dict[str, AgentPool] = {}
        self.load_metrics: Dict[str, float] = {}

    def create_pool(self, name: str, template: BaseAgent, min_instances: int = 1):
        """Create a pool of similar agents"""

        pool = AgentPool(
            name=name,
            template=template,
            instances=[],
            min_instances=min_instances
        )

        # Create initial instances
        for i in range(min_instances):
            instance = self._clone_agent(template, f"{name}-{i}")
            pool.instances.append(instance)
            app.register_agent(instance)

        self.pools[name] = pool

    def _clone_agent(self, template: BaseAgent, new_name: str) -> BaseAgent:
        """Clone an agent with a new name"""

        # Deep copy skills
        cloned_skills = {}
        for skill_name, skill in template.skills.items():
            # Create new skill instance with same config
            skill_class = skill.__class__
            cloned_skills[skill_name] = skill_class(skill.config)

        return BaseAgent(
            name=new_name,
            instructions=template.instructions,
            model=template.model,
            skills=cloned_skills
        )

    async def scale_pool(self, pool_name: str, target_instances: int):
        """Scale pool to target number of instances"""

        if pool_name not in self.pools:
            return

        pool = self.pools[pool_name]
        current = len(pool.instances)

        if target_instances > current:
            # Scale up
            for i in range(current, target_instances):
                instance = self._clone_agent(
                    pool.template, 
                    f"{pool_name}-{i}"
                )
                pool.instances.append(instance)
                app.register_agent(instance)

        elif target_instances < current:
            # Scale down
            for i in range(target_instances, current):
                instance = pool.instances.pop()
                app.unregister_agent(instance.name)

    async def auto_scale(self):
        """Automatically scale pools based on load"""

        while True:
            for pool_name, pool in self.pools.items():
                # Get current load
                load = self.load_metrics.get(pool_name, 0.0)
                current_instances = len(pool.instances)

                # Scale up if load is high
                if load > 0.8 and current_instances < pool.max_instances:
                    await self.scale_pool(pool_name, current_instances + 1)

                # Scale down if load is low
                elif load < 0.3 and current_instances > pool.min_instances:
                    await self.scale_pool(pool_name, current_instances - 1)

            await asyncio.sleep(30)  # Check every 30 seconds

# Usage
manager = DynamicAgentManager()

# Create customer service pool
cs_template = AgentFactory.create_customer_service_agent("premium")
manager.create_pool("customer-service", cs_template, min_instances=2)

# Start auto-scaling
asyncio.create_task(manager.auto_scale())

    Skill-Based Composition

    class ComposableAgent:
    """Agent that can be dynamically composed with different skills"""

    def __init__(self, base_config: Dict):
        self.base_config = base_config
        self.agent = None

    def add_skills(self, skill_configs: Dict[str, Dict]) -> 'ComposableAgent':
        """Add skills to the agent"""

        skills = {}

        for skill_name, config in skill_configs.items():
            skill_class = self._get_skill_class(skill_name)
            skills[skill_name] = skill_class(config)

        # Create or update agent
        if self.agent is None:
            self.agent = BaseAgent(
                name=self.base_config["name"],
                instructions=self.base_config["instructions"],
                model=self.base_config["model"],
                skills=skills
            )
        else:
            # Add skills to existing agent
            for name, skill in skills.items():
                self.agent.add_skill(name, skill)

        return self

    def with_memory(self, memory_type: str = "short_term") -> 'ComposableAgent':
        """Add memory capability"""

        if memory_type == "short_term":
            from robutler.agents.skills import ShortTermMemorySkill
            skill_config = {"memory": {"max_messages": 100}}
        elif memory_type == "long_term":
            from robutler.agents.skills import LongTermMemorySkill
            skill_config = {"memory": {"connection_string": "postgresql://..."}}

        return self.add_skills(skill_config)

    def with_database_access(self, databases: List[str]) -> 'ComposableAgent':
        """Add database access"""

        from robutler.agents.skills import DatabaseSkill
        skill_config = {
            "database": {
                "allowed_databases": databases,
                "read_only": True
            }
        }

        return self.add_skills(skill_config)

    def with_external_apis(self, apis: List[str]) -> 'ComposableAgent':
        """Add external API access"""

        api_skills = {}

        for api in apis:
            if api == "weather":
                from robutler.agents.skills import WeatherSkill
                api_skills["weather"] = {"api_key": "..."}
            elif api == "news":
                from robutler.agents.skills import NewsSkill
                api_skills["news"] = {"api_key": "..."}

        return self.add_skills(api_skills)

    def build(self) -> BaseAgent:
        """Build the final agent"""
        return self.agent

# Usage - Fluent interface
agent = (ComposableAgent({
    "name": "research-assistant",
    "instructions": "You are a research assistant",
    "model": "openai/gpt-4o"
})
.with_memory("long_term")
.with_database_access(["research_db", "papers_db"])
.with_external_apis(["weather", "news"])
.build())

    Feature Flags

    class FeatureFlaggedAgent:
    """Agent with feature flag support"""

    def __init__(self, base_config: Dict, feature_flags: Dict[str, bool]):
        self.base_config = base_config
        self.feature_flags = feature_flags

    def create_agent(self) -> BaseAgent:
        """Create agent based on feature flags"""

        skills = {}
        instructions = [self.base_config["instructions"]]

        # Core skills always included
        from robutler.agents.skills import ShortTermMemorySkill
        skills["memory"] = ShortTermMemorySkill()

        # Feature-gated skills
        if self.feature_flags.get("enable_advanced_search", False):
            from robutler.agents.skills import VectorMemorySkill
            skills["vector_search"] = VectorMemorySkill()
            instructions.append("You have advanced vector search capabilities.")

        if self.feature_flags.get("enable_payments", False):
            from robutler.agents.skills import PaymentSkill
            skills["payments"] = PaymentSkill()
            instructions.append("You can process payments and handle billing.")

        if self.feature_flags.get("enable_collaboration", False):
            from robutler.agents.skills import NLISkill, DiscoverySkill
            skills.update({
                "nli": NLISkill(),
                "discovery": DiscoverySkill()
            })
            instructions.append("You can collaborate with other agents.")

        return BaseAgent(
            name=self.base_config["name"],
            instructions=" ".join(instructions),
            model=self.base_config["model"],
            skills=skills
        )

# Usage with feature flags from config/environment
feature_flags = {
    "enable_advanced_search": os.getenv("ENABLE_VECTOR_SEARCH", "false").lower() == "true",
    "enable_payments": os.getenv("ENABLE_PAYMENTS", "false").lower() == "true",
    "enable_collaboration": os.getenv("ENABLE_COLLABORATION", "true").lower() == "true"
}

agent_creator = FeatureFlaggedAgent({
    "name": "adaptive-assistant",
    "instructions": "You are an adaptive assistant",
    "model": "openai/gpt-4o"
}, feature_flags)

agent = agent_creator.create_agent()

    Agent Templates

    Template System

    from typing import Protocol
from abc import abstractmethod

class AgentTemplate(Protocol):
    """Protocol for agent templates"""

    @abstractmethod
    def create_agent(self, **kwargs) -> BaseAgent:
        """Create agent from template"""
        pass

class CustomerServiceTemplate:
    """Template for customer service agents"""

    def create_agent(
        self,
        name: str,
        tier: str = "basic",
        languages: List[str] = None,
        departments: List[str] = None
    ) -> BaseAgent:

        languages = languages or ["en"]
        departments = departments or ["general"]

        skills = {
            "memory": ShortTermMemorySkill({"max_messages": 100})
        }

        instructions = [
            f"You are a {tier} customer service agent.",
            f"You speak: {', '.join(languages)}",
            f"You handle: {', '.join(departments)}"
        ]

        if tier in ["premium", "enterprise"]:
            from robutler.agents.skills import DiscoverySkill
            skills["discovery"] = DiscoverySkill()
            instructions.append("You can escalate to specialists when needed.")

        return BaseAgent(
            name=name,
            instructions=" ".join(instructions),
            model="openai/gpt-4o",
            skills=skills
        )

class DataAnalystTemplate:
    """Template for data analyst agents"""

    def create_agent(
        self,
        name: str,
        data_sources: List[str],
        analysis_types: List[str] = None
    ) -> BaseAgent:

        analysis_types = analysis_types or ["descriptive", "diagnostic"]

        from robutler.agents.skills import DatabaseSkill

        skills = {
            "database": DatabaseSkill({
                "data_sources": data_sources,
                "allowed_operations": ["SELECT", "WITH"]
            })
        }

        instructions = [
            "You are a data analyst.",
            f"You have access to: {', '.join(data_sources)}",
            f"You specialize in: {', '.join(analysis_types)} analysis"
        ]

        return BaseAgent(
            name=name,
            instructions=" ".join(instructions),
            model="anthropic/claude-3-sonnet",
            skills=skills
        )

# Template registry
AGENT_TEMPLATES = {
    "customer_service": CustomerServiceTemplate(),
    "data_analyst": DataAnalystTemplate()
}

def create_agent_from_template(template_name: str, **kwargs) -> BaseAgent:
    """Create agent from registered template"""

    if template_name not in AGENT_TEMPLATES:
        raise ValueError(f"Unknown template: {template_name}")

    template = AGENT_TEMPLATES[template_name]
    return template.create_agent(**kwargs)

# Usage
cs_agent = create_agent_from_template(
    "customer_service",
    name="support-multilingual",
    tier="premium",
    languages=["en", "es", "fr"],
    departments=["billing", "technical"]
)

    Hot Reloading

    Runtime Agent Updates

    class HotReloadableAgent:
    """Agent that can be updated without downtime"""

    def __init__(self, config_path: str):
        self.config_path = config_path
        self.agent = None
        self.last_modified = 0

    def load_config(self) -> Dict:
        """Load configuration from file"""
        import json
        with open(self.config_path, 'r') as f:
            return json.load(f)

    def get_agent(self) -> BaseAgent:
        """Get agent, reloading if config changed"""

        import os

        # Check if config file was modified
        current_modified = os.path.getmtime(self.config_path)

        if current_modified > self.last_modified:
            self.reload_agent()
            self.last_modified = current_modified

        return self.agent

    def reload_agent(self):
        """Reload agent from updated configuration"""

        config = self.load_config()

        # Create new agent
        new_agent = BaseAgent(
            name=config["name"],
            instructions=config["instructions"],
            model=config["model"],
            skills=self._create_skills(config.get("skills", {}))
        )

        # Replace old agent
        if self.agent:
            app.unregister_agent(self.agent.name)

        self.agent = new_agent
        app.register_agent(new_agent)

    def _create_skills(self, skill_configs: Dict) -> Dict:
        """Create skills from configuration"""
        skills = {}

        for skill_name, skill_config in skill_configs.items():
            skill_class = self._get_skill_class(skill_name)
            skills[skill_name] = skill_class(skill_config)

        return skills

# Usage
reloadable_agent = HotReloadableAgent("agent_config.json")

# The agent will automatically reload when config.json changes
@app.middleware("http")
async def agent_reload_middleware(request, call_next):
    # Ensure agent is up to date
    reloadable_agent.get_agent()
    return await call_next(request)
