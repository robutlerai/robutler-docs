    Core Features

    The Robutler Python SDK provides four main components:

    • üéØ Skills System: Modular capabilities (LLM, memory, discovery, payments, MCP)
    • ü§ù Agent Discovery: Find and communicate with other agents
    • üí≥ Payment Integration: Built-in credit system and billing

    Basic Agent

    from robutler.agents.core.base_agent import BaseAgent
from robutler.agents.skills.core.llm.openai.skill import OpenAISkill

# Create agent with LLM skill
agent = BaseAgent(
    name="assistant",
    instructions="You are a helpful AI assistant.",
    skills={"llm": OpenAISkill({"model": "gpt-4o-mini"})}
)

# Run chat completion
messages = [{"role": "user", "content": "Hello!"}]
response = await agent.run(messages=messages)
print(response.content)

    Smart Model Parameter

    Use the model parameter for quick LLM skill creation:

    # Automatically creates OpenAI skill with GPT-4o
agent = BaseAgent(
    name="assistant",
    instructions="You are a helpful assistant.",
    model="openai/gpt-4o"
)

# Other providers
assistant_anthropic = BaseAgent(model="anthropic/claude-3-sonnet")
assistant_litellm = BaseAgent(model="litellm/gpt-4o-mini")  
assistant_xai = BaseAgent(model="xai/grok-beta")

    Skills System

    Memory Skills

    from robutler.agents.skills.core.memory.short_term.skill import ShortTermMemorySkill
from robutler.agents.skills.core.memory.vector_memory.skill import VectorMemorySkill

agent = BaseAgent(
    name="memory-agent",
    model="openai/gpt-4o-mini",
    skills={
        "llm": OpenAISkill({"model": "gpt-4o-mini"}),
        "memory": ShortTermMemorySkill({"max_tokens": 4000}),
        "vector": VectorMemorySkill({"collection": "documents"})
    }
)

# Memory skills automatically filter messages and manage context
messages = [
    {"role": "user", "content": "Remember: my name is Alex"},
    {"role": "assistant", "content": "I'll remember that your name is Alex."},
    {"role": "user", "content": "What's my name?"}
]
response = await agent.run(messages=messages)

    Payment Integration

    from robutler.agents.skills.robutler.payments import PaymentSkill
from robutler.agents.skills.robutler.auth.skill import AuthSkill

agent = BaseAgent(
    name="premium-agent",
    model="openai/gpt-4o-mini",
    skills={
        "llm": OpenAISkill({"model": "gpt-4o-mini"}),
        "auth": AuthSkill(),
        "payments": PaymentSkill({
            "enable_billing": True,
            "agent_pricing_percent": 20,
            "minimum_balance": 1.0
        })
    }
)

# Usage is logged centrally to context.usage. On finalize, the token is charged.
response = await agent.run(messages=[
    {"role": "user", "content": "Analyze this complex dataset"}
])

    Custom Tools

    from robutler.agents.tools.decorators import tool, http

@tool
def calculate_metrics(data: str) -> str:
    """Calculate business metrics from data."""
    # Custom analysis logic
    return "Metrics: Revenue +15%, Users +8%"

@http("/weather", method="get", scope="owner")
def get_weather(location: str, units: str = "celsius") -> dict:
    """Weather API endpoint"""
    return {
        "location": location,
        "temperature": 25,
        "units": units,
        "condition": "sunny"
    }

@http("/metrics", method="get")
def get_metrics_api() -> dict:
    """Business metrics API endpoint"""
    return {
        "revenue": "+15%",
        "users": "+8%",
        "satisfaction": "95%"
    }

# Tools and HTTP endpoints are automatically registered
agent = BaseAgent(
    name="analyst",
    model="openai/gpt-4o-mini",
    instructions="You analyze business metrics using available tools.",
    tools=[calculate_metrics],              # Internal tools for LLM
    http_handlers=[get_weather, get_metrics_api]  # HTTP API endpoints
)

# LLM can use internal tools
response = await agent.run(messages=[
    {"role": "user", "content": "Calculate metrics for Q4 sales data"}
])

# HTTP endpoints available at:
# GET /analyst/weather?location=NYC&units=fahrenheit
# GET /analyst/metrics

    Capabilities Auto-Registration

    Use the capabilities parameter to automatically register any decorated functions:

    from robutler.agents.tools.decorators import tool, http, hook, handoff

@tool(scope="owner")
def my_tool(data: str) -> str:
    return f"Processed: {data}"

@http("/api/process")
def my_api(input: dict) -> dict:
    return {"processed": input}

@hook("on_request", priority=10)
def my_hook(context):
    print("Request received")
    return context

# Auto-register all decorated functions
agent = BaseAgent(
    name="capable-agent",
    model="openai/gpt-4o-mini",
    capabilities=[my_tool, my_api, my_hook]  # Automatically categorized
)

    HTTP Server

    Deploy agents as OpenAI-compatible API:

    from robutler.server.core.app import create_server

# Create agents
agents = [
    BaseAgent(name="assistant", model="openai/gpt-4o-mini"),
    BaseAgent(name="analyst", model="anthropic/claude-3-sonnet")
]

# Create server
server = create_server(
    agents=agents,
    enable_monitoring=True,
    enable_dynamic_agents=True
)

# Run with uvicorn
import uvicorn
uvicorn.run(server.app, host="0.0.0.0", port=8000)

])
# Credits are automatically deducted based on usage
