    Skill Tools

    Tools are executable functions that extend agent capabilities. The @tool decorator can be used in skills or on standalone functions.

    When used with PaymentSkill, priced tools should return (result, usage_payload) where usage_payload contains a pricing object. The agent records this to context.usage, and the payment skill charges at finalize_connection.

    Tool Definition Methods

    Tools in Skills

    from robutler.agents.skills import Skill
from robutler.agents.tools.decorators import tool

class CalculatorSkill(Skill):
    @tool
    def add(self, a: float, b: float) -> float:
        """Add two numbers together"""
        return a + b

    @tool
    def multiply(self, x: float, y: float) -> float:
        """Multiply two numbers"""
        return x * y

    Standalone Tool Functions

    The @tool decorator can also be used on standalone functions outside of skills:

    from robutler.agents.tools.decorators import tool

@tool
def calculate_tax(amount: float, rate: float = 0.08) -> float:
    """Calculate tax on an amount"""
    return amount * rate

@tool(scope="owner")
def admin_reset(system: str) -> str:
    """Reset system (admin only)"""
    return f"System {system} has been reset"

# These can be passed to agents directly
from robutler.agents import BaseAgent

agent = BaseAgent(
    name="calculator",
    model="openai/gpt-4o",
    tools=[calculate_tax, admin_reset]  # Standalone tools
)

    Tool with Complex Parameters

    class DataSkill(Skill):
    @tool
    def analyze_data(
        self,
        data: List[float],
        method: str = "mean",
        precision: int = 2
    ) -> Dict[str, Any]:
        """Analyze numerical data

        Args:
            data: List of numbers to analyze
            method: Analysis method (mean, median, std)
            precision: Decimal places for rounding

        Returns:
            Analysis results
        """
        if method == "mean":
            result = sum(data) / len(data)
        elif method == "median":
            sorted_data = sorted(data)
            n = len(sorted_data)
            result = sorted_data[n//2] if n % 2 else (sorted_data[n//2-1] + sorted_data[n//2]) / 2
        elif method == "std":
            mean = sum(data) / len(data)
            variance = sum((x - mean) ** 2 for x in data) / len(data)
            result = variance ** 0.5
        else:
            return {"error": f"Unknown method: {method}"}

        return {
            "method": method,
            "result": round(result, precision),
            "data_points": len(data)
        }

    Tool Registration

    Automatic Registration

    Tools decorated with @tool are automatically registered when the skill is initialized:

    class AutoSkill(Skill):
    @tool  # Automatically registered
    def auto_tool(self, param: str) -> str:
        return f"Processed: {param}"

    Manual Registration

    For dynamic tools or conditional registration:

    class DynamicSkill(Skill):
    async def initialize(self, agent):
        await super().initialize(agent)

        # Conditionally register tools
        if self.config.get("advanced_mode"):
            self.register_tool(self.advanced_tool)

    def advanced_tool(self, data: str) -> str:
        """Dynamically registered advanced tool"""
        return f"Advanced processing: {data}"

    Runtime Registration

    Register tools during request processing:

    class AdaptiveSkill(Skill):
    @hook("on_connection")
    async def adapt_tools(self, context):
        """Register tools based on user context"""

        user_level = context.get("user_level", "basic")

        if user_level == "expert":
            self.register_tool(self.expert_analysis)
            self.register_tool(self.bulk_operations)

        if context.get("beta_features"):
            self.register_tool(self.experimental_tool)

        return context

    Tool Scoping

    Access Control

    class SecureSkill(Skill):
    @tool(scope="all")  # Available to everyone
    def public_info(self) -> str:
        return "Public information"

    @tool(scope="owner")  # Agent owner only
    def owner_config(self, setting: str, value: str) -> str:
        return f"Updated {setting} to {value}"

    @tool(scope="admin")  # System admins only
    def admin_debug(self) -> Dict:
        return {
            "memory": self.get_memory_usage(),
            "connections": self.get_active_connections()
        }

    Dynamic Scoping

    class DynamicSecuritySkill(Skill):
    @tool
    def sensitive_operation(self, action: str) -> str:
        """Tool with runtime scope checking"""

        context = self.get_context()
        user_role = context.get("user_role", "user")

        if action == "read" and user_role in ["user", "owner", "admin"]:
            return "Read operation allowed"
        elif action == "write" and user_role in ["owner", "admin"]:
            return "Write operation allowed"
        elif action == "delete" and user_role == "admin":
            return "Delete operation allowed"
        else:
            return f"Permission denied for {action}"

    Async Tools

    Async Tool Definition

    class AsyncSkill(Skill):
    @tool
    async def fetch_data(self, url: str) -> Dict:
        """Fetch data from external API"""
        import aiohttp

        async with aiohttp.ClientSession() as session:
            async with session.get(url) as response:
                return await response.json()

    @tool
    async def parallel_process(self, items: List[str]) -> List[str]:
        """Process multiple items in parallel"""
        tasks = [self.process_item(item) for item in items]
        results = await asyncio.gather(*tasks)
        return results

    async def process_item(self, item: str) -> str:
        # Simulate async processing
        await asyncio.sleep(0.1)
        return item.upper()

    class RobustSkill(Skill):
    @tool
    def safe_divide(self, a: float, b: float) -> Dict[str, Any]:
        """Division with error handling"""
        try:
            if b == 0:
                return {
                    "success": False,
                    "error": "Division by zero",
                    "suggestion": "Please provide a non-zero divisor"
                }

            result = a / b
            return {
                "success": True,
                "result": result,
                "calculation": f"{a} Ã· {b} = {result}"
            }

        except Exception as e:
            return {
                "success": False,
                "error": str(e),
                "suggestion": "Please check your inputs"
            }

    Validation

    class ValidatedSkill(Skill):
    @tool
    def create_user(
        self,
        username: str,
        email: str,
        age: int
    ) -> Dict[str, Any]:
        """Create user with validation"""

        # Validate inputs
        errors = []

        if not username or len(username) < 3:
            errors.append("Username must be at least 3 characters")

        if "@" not in email:
            errors.append("Invalid email format")

        if age < 18:
            errors.append("Must be 18 or older")

        if errors:
            return {
                "success": False,
                "errors": errors
            }

        # Create user
        user = {
            "id": self.generate_id(),
            "username": username,
            "email": email,
            "age": age,
            "created": datetime.now().isoformat()
        }

        return {
            "success": True,
            "user": user
        }

    Tool Pipelines

    class PipelineSkill(Skill):
    @tool
    async def process_pipeline(
        self,
        input_data: str,
        steps: List[str]
    ) -> Dict:
        """Process data through multiple steps"""

        result = input_data
        history = [{"step": "input", "data": input_data}]

        for step in steps:
            if step == "clean":
                result = self.clean_data(result)
            elif step == "transform":
                result = self.transform_data(result)
            elif step == "validate":
                result = self.validate_data(result)
            else:
                return {
                    "error": f"Unknown step: {step}",
                    "history": history
                }

            history.append({"step": step, "data": result})

        return {
            "final_result": result,
            "history": history,
            "steps_completed": len(steps)
        }

    Tool Decorators

    Custom Decorators

    from functools import wraps

def rate_limit(calls_per_minute: int = 10):
    """Rate limit decorator for tools"""
    def decorator(func):
        func._rate_limit = calls_per_minute
        return func
    return decorator

def cache_result(ttl_seconds: int = 300):
    """Cache decorator for tools"""
    def decorator(func):
        cache = {}

        @wraps(func)
        def wrapper(*args, **kwargs):
            key = str(args) + str(kwargs)

            if key in cache:
                result, timestamp = cache[key]
                if time.time() - timestamp < ttl_seconds:
                    return result

            result = func(*args, **kwargs)
            cache[key] = (result, time.time())
            return result

        return wrapper
    return decorator

class OptimizedSkill(Skill):
    @tool
    @rate_limit(calls_per_minute=30)
    @cache_result(ttl_seconds=600)
    def expensive_calculation(self, n: int) -> int:
        """Rate-limited and cached calculation"""
        # Simulate expensive operation
        return sum(i ** 2 for i in range(n))

    class DocumentedSkill(Skill):
    @tool(
        name="advanced_search",
        description="Search with advanced filters and options",
        metadata={
            "category": "search",
            "version": "2.0",
            "deprecated": False,
            "examples": [
                {"query": "python tutorial", "filters": {"type": "video"}},
                {"query": "machine learning", "filters": {"date": "2024"}}
            ]
        }
    )
    def search_with_filters(
        self,
        query: str,
        filters: Dict[str, Any] = None
    ) -> List[Dict]:
        """Advanced search implementation"""
        results = self.basic_search(query)

        if filters:
            results = self.apply_filters(results, filters)

        return results

    Testing Tools

    import pytest
from robutler.agents import BaseAgent

class TestCalculatorSkill:
    @pytest.fixture
    def skill(self):
        return CalculatorSkill()

    @pytest.fixture
    def agent(self, skill):
        return BaseAgent(
            name="test-agent",
            model="openai/gpt-4o",
            skills={"calc": skill}
        )

    def test_add_tool(self, skill):
        """Test add tool directly"""
        result = skill.add(2, 3)
        assert result == 5

    @pytest.mark.asyncio
    async def test_tool_discovery(self, agent):
        """Test that tools are discovered"""
        tools = agent.get_all_tools()
        tool_names = [t["function"]["name"] for t in tools]

        assert "add" in tool_names
        assert "multiply" in tool_names

    @pytest.mark.asyncio
    async def test_tool_execution(self, agent):
        """Test tool execution through agent"""
        response = await agent.run([
            {"role": "user", "content": "Calculate 7 * 8"}
        ])

        # Should use multiply tool
        assert "56" in response.choices[0].message.content

    Best Practices

    1. Clear Naming - Use descriptive tool names
    2. Type Hints - Always include type hints for auto-schema
    3. Docstrings - Provide clear descriptions and examples
    4. Error Handling - Return errors as data, not exceptions
    5. Validation - Validate inputs before processing
    6. Async When Needed - Use async for I/O operations
    7. Appropriate Scoping - Set correct access levels
