    Skill Prompts

    Prompts allow skills to guide LLM behavior with specific instructions, context, and constraints.

    Prompts are collected from all active skills and merged into the system message. Keep prompts short, declarative, and focused on behavior that cannot be inferred from tools alone.

    Understanding Prompts

    Skill prompts are instructions that get injected into the system message, helping the LLM understand:

    • What capabilities the skill provides
    • How to use skill tools effectively
    • Domain-specific knowledge and constraints
    • Behavioral guidelines

    Basic Prompt Implementation

    from robutler.agents.skills import Skill

class WeatherSkill(Skill):
    def get_prompts(self) -> List[str]:
        """Return skill-specific prompts"""
        return [
            "You have access to real-time weather information.",
            "Always specify the location when discussing weather.",
            "Include both Celsius and Fahrenheit in temperature responses.",
            "If no location is specified, ask the user for their location."
        ]

    Context-Aware Prompts

    class AdaptiveSkill(Skill):
    def get_prompts(self) -> List[str]:
        """Generate prompts based on context"""
        prompts = ["You have adaptive capabilities."]

        # Access context if available
        context = self.get_context()
        if context:
            user_type = context.get("user_type", "standard")

            if user_type == "technical":
                prompts.append("Provide detailed technical explanations.")
                prompts.append("Include code examples when relevant.")
            else:
                prompts.append("Explain concepts in simple terms.")
                prompts.append("Avoid technical jargon.")

        return prompts

    Configuration-Based Prompts

    class ConfigurableSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config)
        self.style = config.get("style", "professional")
        self.verbosity = config.get("verbosity", "normal")

    def get_prompts(self) -> List[str]:
        """Generate prompts from configuration"""
        prompts = []

        # Style prompts
        if self.style == "casual":
            prompts.append("Use a friendly, conversational tone.")
        elif self.style == "formal":
            prompts.append("Maintain a formal, professional tone.")

        # Verbosity prompts
        if self.verbosity == "concise":
            prompts.append("Keep responses brief and to the point.")
        elif self.verbosity == "detailed":
            prompts.append("Provide comprehensive, detailed responses.")

        return prompts

    Advanced Prompt Patterns

    Progressive Disclosure

    class TeachingSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config)
        self.detail_level = 1

    def get_prompts(self) -> List[str]:
        base_prompts = [
            "You are an adaptive teacher.",
            "Start with simple explanations."
        ]

        if self.detail_level >= 2:
            base_prompts.append("Include examples and analogies.")

        if self.detail_level >= 3:
            base_prompts.append("Provide technical details and edge cases.")

        return base_prompts

    @hook("on_message")
    async def adjust_detail(self, context):
        """Adjust detail level based on user responses"""
        message = context.messages[-1]["content"]

        if "more detail" in message.lower():
            self.detail_level = min(self.detail_level + 1, 3)
        elif "simpler" in message.lower():
            self.detail_level = max(self.detail_level - 1, 1)

        return context

    Contextual Examples

    class ExampleSkill(Skill):
    def get_prompts(self) -> List[str]:
        examples = self.get_relevant_examples()

        prompts = [
            "Provide examples to illustrate concepts.",
            "Use these example formats:"
        ]

        for example in examples:
            prompts.append(f"- {example['pattern']}: {example['usage']}")

        return prompts

    def get_relevant_examples(self) -> List[Dict]:
        """Get examples based on current context"""
        context = self.get_context()
        if not context:
            return []

        # Return domain-specific examples
        domain = context.get("domain", "general")
        return self.examples_db.get(domain, [])

    Testing Prompts

    import pytest
from robutler.agents import BaseAgent

class TestSkillPrompts:
    @pytest.mark.asyncio
    async def test_prompt_injection(self):
        """Test that prompts are properly injected"""

        skill = WeatherSkill()
        agent = BaseAgent(
            name="test-agent",
            model="openai/gpt-4o",
            skills={"weather": skill}
        )

        # Get system message
        system_message = agent.get_system_message()

        # Verify prompts are included
        for prompt in skill.get_prompts():
            assert prompt in system_message

    @pytest.mark.asyncio
    async def test_dynamic_prompts(self):
        """Test context-aware prompt generation"""

        skill = AdaptiveSkill()

        # Test without context
        prompts1 = skill.get_prompts()
        assert len(prompts1) == 3  # Base prompts

        # Test with context
        with Context(user_type="technical"):
            prompts2 = skill.get_prompts()
            assert len(prompts2) == 5  # Base + technical prompts

    Best Practices

    1. Be Specific - Clear, actionable instructions work best
    2. Use Examples - Show don't just tell
    3. Set Boundaries - Define what the skill should and shouldn't do
    4. Progressive Complexity - Start simple, add detail as needed
    5. Test Prompts - Verify they produce desired behavior
    6. Version Prompts - Track changes as you refine them

    Common Patterns

    DO Pattern

    [
    "DO: Always validate user input",
    "DO: Provide helpful error messages",
    "DO: Ask for clarification when needed"
]

    DON'T Pattern

    [
    "DON'T: Make assumptions about user intent",
    "DON'T: Execute dangerous operations",
    "DON'T: Share sensitive information"
]
