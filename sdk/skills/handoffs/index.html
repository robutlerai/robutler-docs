    Skill Handoffs

    Handoffs enable skills to route queries to specialized agents when they need expertise beyond their scope.

    Understanding Handoffs

    Handoffs allow agents to: - Route complex queries to specialized agents - Maintain context across agent transitions - Provide seamless multi-agent workflows - Enable natural language collaboration

    Defining Handoffs in Skills

    Basic Handoff

    from robutler.agents.skills import Skill
from robutler.agents.skills.decorators import handoff

class GeneralSkill(Skill):
    @handoff("finance-expert")
    def needs_finance_expert(self, query: str) -> bool:
        """Route finance questions to specialist"""
        finance_terms = ["stock", "investment", "portfolio", "trading", "finance"]
        return any(term in query.lower() for term in finance_terms)

    @handoff("legal-advisor")
    def needs_legal_expert(self, query: str) -> bool:
        """Route legal questions to specialist"""
        legal_terms = ["contract", "legal", "law", "compliance", "regulation"]
        return any(term in query.lower() for term in legal_terms)

    Dynamic Handoff

    class RouterSkill(Skill):
    @handoff()  # No fixed target
    def route_to_specialist(self, query: str) -> str:
        """Dynamically determine target agent"""

        # Analyze query complexity and topic
        if self.is_medical_query(query):
            return "medical-assistant"
        elif self.is_technical_query(query):
            return "tech-support"
        elif self.is_creative_query(query):
            return "creative-writer"

        return None  # No handoff needed

    Conditional Handoffs

    class SmartRouterSkill(Skill):
    @handoff("premium-expert")
    def needs_premium_expert(self, query: str) -> bool:
        """Route to premium expert based on user tier"""

        context = self.get_context()
        user_tier = context.get("user_tier", "basic")

        # Only route premium users to premium experts
        if user_tier != "premium":
            return False

        return self.is_complex_query(query)

    Adding Context for Target Agent

    class ContextualHandoffSkill(Skill):
    @handoff("data-analyst")
    def needs_analysis(self, query: str) -> bool:
        return "analyze" in query or "data" in query

    @hook("before_handoff")
    async def prepare_analysis_context(self, context):
        """Add context for data analyst"""

        if context["handoff_agent"] == "data-analyst":
            # Extract data requirements
            data_context = {
                "data_sources": self.identify_data_sources(context.messages),
                "analysis_type": self.detect_analysis_type(context.messages[-1]["content"]),
                "user_background": self.get_user_background(context.peer_user_id),
                "conversation_summary": self.summarize_conversation(context.messages)
            }

            context["handoff_metadata"] = data_context

        return context

    Multi-Step Workflows

    Sequential Handoffs

    class WorkflowSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config)
        self.workflow_state = {}

    @handoff("researcher")
    def needs_research(self, query: str) -> bool:
        """First step: research phase"""
        workflow_id = self.get_context().completion_id

        if workflow_id not in self.workflow_state:
            self.workflow_state[workflow_id] = {"step": "research"}
            return "research" in query or "find information" in query

        return False

    @handoff("writer")
    def needs_writing(self, query: str) -> bool:
        """Second step: writing phase"""
        workflow_id = self.get_context().completion_id
        state = self.workflow_state.get(workflow_id, {})

        if state.get("step") == "research" and state.get("research_complete"):
            self.workflow_state[workflow_id]["step"] = "writing"
            return True

        return False

    @hook("after_handoff")
    async def track_workflow_progress(self, context):
        """Track workflow state transitions"""
        workflow_id = context.completion_id
        target_agent = context["handoff_agent"]

        if target_agent == "researcher":
            self.workflow_state[workflow_id]["research_complete"] = True
        elif target_agent == "writer":
            self.workflow_state[workflow_id]["writing_complete"] = True

        return context

    Parallel Consultation

    class ConsultationSkill(Skill):
    @tool
    async def multi_expert_consultation(self, question: str) -> Dict:
        """Consult multiple experts in parallel"""

        # Identify relevant experts
        experts = self.identify_experts(question)

        # Query all experts concurrently
        tasks = []
        for expert in experts:
            task = self.nli.query_agent(expert, question)
            tasks.append(task)

        # Gather responses
        responses = await asyncio.gather(*tasks, return_exceptions=True)

        # Synthesize expert opinions
        synthesis = {}
        for expert, response in zip(experts, responses):
            if isinstance(response, Exception):
                synthesis[expert] = "Unavailable"
            else:
                synthesis[expert] = response.get("response", "No response")

        return {
            "question": question,
            "expert_opinions": synthesis,
            "consensus": self.find_consensus(synthesis),
            "conflicts": self.identify_conflicts(synthesis)
        }

    Platform Integration

    Using NLI Skill

    from robutler.agents.skills import NLISkill

class NLIHandoffSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config, dependencies=["nli"])

    @handoff()
    async def intelligent_routing(self, query: str) -> str:
        """Use NLI to find best agent"""

        # Get NLI skill from dependencies
        nli = self.agent.skills.get("nli")
        if not nli:
            return None

        # Query for suitable agents
        agents = await nli.find_capable_agents(
            query=query,
            max_results=3
        )

        if not agents:
            return None

        # Select best match
        best_agent = max(agents, key=lambda a: a.get("confidence", 0))

        return best_agent["name"] if best_agent["confidence"] > 0.7 else None

    Discovery-Based Routing

    from robutler.agents.skills import DiscoverySkill

class DiscoveryHandoffSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config, dependencies=["discovery"])

    @handoff()
    async def discover_and_route(self, query: str) -> str:
        """Discover agents and route to best match"""

        discovery = self.agent.skills.get("discovery")
        if not discovery:
            return None

        # Find agents with matching capabilities
        agents = await discovery.find_agents(
            intent=query,
            skills=self.extract_required_skills(query),
            max_results=5
        )

        # Score agents based on match quality
        scored_agents = []
        for agent in agents:
            score = self.calculate_match_score(query, agent)
            scored_agents.append((score, agent))

        # Return best match if score is high enough
        if scored_agents:
            best_score, best_agent = max(scored_agents)
            if best_score > 0.8:
                return best_agent["name"]

        return None

    Error Handling

    Fallback Strategies

    class ResilientHandoffSkill(Skill):
    @handoff("primary-expert", fallback="general-assistant")
    def needs_expert_with_fallback(self, query: str) -> bool:
        """Handoff with automatic fallback"""
        return "expert help" in query

    @hook("after_handoff")
    async def handle_handoff_failure(self, context):
        """Handle failed handoffs gracefully"""

        handoff_result = context["handoff_result"]

        if not handoff_result.get("success"):
            # Try fallback agent
            fallback_agent = self.get_fallback_agent(context["handoff_agent"])

            if fallback_agent:
                fallback_result = await self.nli.query_agent(
                    fallback_agent,
                    context.messages[-1]["content"]
                )

                if fallback_result.get("success"):
                    context["handoff_result"] = fallback_result
                    context["used_fallback"] = True
                else:
                    # Provide local response as last resort
                    context["handoff_result"] = {
                        "success": True,
                        "response": self.generate_local_response(context.messages[-1]["content"]),
                        "local_fallback": True
                    }

        return context

    Timeout Handling

    class TimeoutHandoffSkill(Skill):
    @handoff("slow-expert", timeout=30)
    def needs_slow_expert(self, query: str) -> bool:
        """Handoff with timeout protection"""
        return "complex analysis" in query

    @hook("before_handoff")
    async def set_timeout(self, context):
        """Set timeout for handoff"""
        context["handoff_timeout"] = 30
        return context

    @hook("after_handoff")
    async def handle_timeout(self, context):
        """Handle handoff timeouts"""

        if context["handoff_result"].get("error") == "timeout":
            # Provide timeout response
            context["handoff_result"] = {
                "success": True,
                "response": "The expert is taking longer than expected. I'll provide what I can help with directly.",
                "timeout_fallback": True
            }

        return context

    Best Practices

    Clear Handoff Conditions

    class WellDefinedHandoffSkill(Skill):
    @handoff("finance-expert")
    def needs_finance_expert(self, query: str) -> bool:
        """Clear, testable handoff condition"""

        # Specific domain indicators
        finance_indicators = [
            "investment", "portfolio", "stock", "bond", "mutual fund",
            "401k", "retirement", "tax planning", "financial planning"
        ]

        # Check for explicit requests
        if any(indicator in query.lower() for indicator in finance_indicators):
            return True

        # Check for complex financial calculations
        if self.contains_financial_calculation(query):
            return True

        # Check for regulatory questions
        if self.mentions_financial_regulations(query):
            return True

        return False

    Context Preservation

    class ContextPreservingSkill(Skill):
    @hook("before_handoff")
    async def preserve_context(self, context):
        """Ensure important context is passed to target agent"""

        # Extract conversation context
        conversation_context = {
            "user_preferences": self.get_user_preferences(context.peer_user_id),
            "conversation_history": self.summarize_conversation(context.messages),
            "previous_actions": self.get_user_actions(context.peer_user_id),
            "current_state": self.get_session_state(context.completion_id)
        }

        # Add domain-specific context
        if context["handoff_agent"] == "finance-expert":
            conversation_context["risk_tolerance"] = self.assess_risk_tolerance(context.messages)
            conversation_context["investment_goals"] = self.extract_investment_goals(context.messages)

        context["handoff_context"] = conversation_context

        return context

from unittest.mock import AsyncMock

class TestHandoffSkill:
    @pytest.fixture
    def skill(self):
        return GeneralSkill()

    def agent(self, skill):
        from robutler.agents import BaseAgent
        from robutler.agents.skills import NLISkill

            name="test-agent",
            model="openai/gpt-4o",

                "general": skill,
                "nli": NLISkill()

    def test_finance_handoff_condition(self, skill):
        """Test finance handoff condition"""

        # Should trigger handoff
        assert skill.needs_finance_expert("Help me with my investment portfolio")
        assert skill.needs_finance_expert("What stocks should I buy?")

        # Should not trigger handoff
        assert not skill.needs_finance_expert("What's the weather like?")
        assert not skill.needs_finance_expert("Tell me a joke")

    @pytest.mark.asyncio
    async def test_handoff_execution(self, agent):
        """Test actual handoff execution"""

        # Mock NLI skill
        nli_skill = agent.skills["nli"]
        nli_skill.query_agent = AsyncMock(return_value={
            "success": True,
            "response": "Based on your investment goals..."
        })

        # Test query that should trigger handoff
        response = await agent.run([
            {"role": "user", "content": "Help me analyze my stock portfolio"}
        ])

        # Verify handoff was triggered
        nli_skill.query_agent.assert_called_once()

        # Verify response includes expert knowledge
        assert "investment" in response.choices[0].message.content.lower()

    @pytest.mark.asyncio
    async def test_handoff_failure_handling(self, agent):
        """Test graceful handling of handoff failures"""

        # Mock NLI skill to fail
        nli_skill = agent.skills["nli"]
        nli_skill.query_agent = AsyncMock(return_value={
            "success": False,
            "error": "Expert unavailable"

        # Test that agent handles failure gracefully
        response = await agent.run([

        # Should still get a response, not an error
        assert response.choices[0].message.content
        assert "error" not in response.choices[0].message.content.lower()

    Complete Example

    from robutler.agents import BaseAgent
from robutler.agents.skills import Skill, NLISkill, DiscoverySkill
from robutler.agents.skills.decorators import handoff, hook

class CustomerServiceSkill(Skill):
    def __init__(self, config=None):
        super().__init__(config, dependencies=["nli", "discovery"])

    @handoff()
    def route_to_department(self, query: str) -> str:
        """Route to appropriate department"""
        query_lower = query.lower()

        # Billing department
        if any(word in query_lower for word in ["bill", "payment", "charge", "refund"]):
            return "billing-department"

        # Technical support
        if any(word in query_lower for word in ["broken", "error", "bug", "technical"]):
            return "technical-support" 

        # Sales department
        if any(word in query_lower for word in ["buy", "purchase", "pricing", "upgrade"]):
            return "sales-team"

        return None

    @hook("before_handoff")
    async def add_customer_context(self, context):
        """Add customer information before handoff"""

        customer_id = context.peer_user_id

        # Get customer data
        customer_data = await self.get_customer_data(customer_id)

        # Add comprehensive context
        context["handoff_metadata"] = {
            "customer_tier": customer_data.get("tier", "standard"),
            "account_status": customer_data.get("status", "active"),
            "previous_tickets": customer_data.get("recent_tickets", []),
            "purchase_history": customer_data.get("recent_purchases", []),
            "conversation_summary": self.summarize_conversation(context.messages)
        }

        return context

    @hook("after_handoff")
    async def track_resolution(self, context):
        """Track handoff outcomes"""

        handoff_result = context["handoff_result"]

        if handoff_result.get("success"):
            # Log successful handoff
            await self.log_successful_handoff(
                department=context["handoff_agent"],
                customer=context.peer_user_id,
                resolution=handoff_result.get("resolution_type")
            )
        else:
            # Create escalation ticket
            await self.create_escalation_ticket(
                customer=context.peer_user_id,
                failed_department=context["handoff_agent"],
                original_query=context.messages[-1]["content"]
            )

        return context

# Create customer service agent
agent = BaseAgent(
    name="customer-service",
    instructions="You are a customer service coordinator. Route queries to appropriate departments and ensure customer satisfaction.",
    model="openai/gpt-4o",
    skills={
        "routing": CustomerServiceSkill(),
        "nli": NLISkill(),
        "discovery": DiscoverySkill()
    }
)
