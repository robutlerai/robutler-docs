    Build your own skills to extend agent capabilities with domain-specific functionality.

    Skill Structure

    Basic Skill Template

    from robutler.agents.skills import Skill
from robutler.agents.tools.decorators import tool
from robutler.agents.skills.decorators import hook, handoff
from typing import List, Dict, Any

class MyCustomSkill(Skill):
    def __init__(self, config: Dict[str, Any] = None):
        super().__init__(
            config=config,
            scope="all",  # "all", "owner", "admin"
            dependencies=[]  # List of required skills
        )

        # Initialize skill-specific state
        self.api_key = config.get("api_key") if config else None
        self.custom_data = {}

    async def initialize(self, agent):
        """Called when skill is added to agent"""
        await super().initialize(agent)

        # Perform initialization
        self.setup_connections()
        self.load_configuration()

    def get_prompts(self) -> List[str]:
        """Provide skill-specific prompts for the LLM"""
        return [
            "You have access to custom functionality.",
            "Use the custom tools when appropriate.",
            "Always validate inputs before processing."
        ]

    @tool
    def custom_function(self, param: str) -> str:
        """Custom tool implementation"""
        return f"Processed: {param}"

    @hook("on_message")
    async def process_message(self, context):
        """Process each message"""
        # Custom logic here
        return context

    @handoff("specialist-agent")
    def needs_specialist(self, query: str) -> bool:
        """Determine if specialist is needed"""
        return "complex" in query.lower()

    Service Integration Skill

    import aiohttp
from typing import Optional

class WeatherSkill(Skill):
    """Integrate with weather API service"""

    def __init__(self, config: Dict[str, Any] = None):
        super().__init__(config, scope="all")

        self.api_key = config.get("api_key")
        self.base_url = config.get("base_url", "https://api.openweathermap.org/data/2.5")
        self.session: Optional[aiohttp.ClientSession] = None

    async def initialize(self, agent):
        await super().initialize(agent)

        # Create HTTP session
        self.session = aiohttp.ClientSession()

        # Validate API key
        if not self.api_key:
            raise ValueError("Weather API key required")

    def get_prompts(self) -> List[str]:
        return [
            "You have access to real-time weather data.",
            "Use get_weather when users ask about weather conditions.",
            "Always include temperature in both Celsius and Fahrenheit.",
            "If location is unclear, ask the user to specify."
        ]

    @tool
    async def get_weather(self, location: str) -> Dict[str, Any]:
        """Get current weather for a location"""

        if not self.session:
            return {"error": "Weather service not initialized"}

        try:
            url = f"{self.base_url}/weather"
            params = {
                "q": location,
                "appid": self.api_key,
                "units": "metric"
            }

            async with self.session.get(url, params=params) as response:
                if response.status == 200:
                    data = await response.json()

                    return {
                        "location": data["name"],
                        "temperature_c": data["main"]["temp"],
                        "temperature_f": (data["main"]["temp"] * 9/5) + 32,
                        "description": data["weather"][0]["description"],
                        "humidity": data["main"]["humidity"],
                        "pressure": data["main"]["pressure"]
                    }
                else:
                    return {"error": f"Weather API error: {response.status}"}

        except Exception as e:
            return {"error": f"Failed to fetch weather: {str(e)}"}

    @tool
    async def get_forecast(self, location: str, days: int = 5) -> List[Dict]:
        """Get weather forecast for multiple days"""

        try:
            url = f"{self.base_url}/forecast"
            params = {
                "q": location,
                "appid": self.api_key,
                "units": "metric",
                "cnt": days * 8  # 8 forecasts per day (3-hour intervals)
            }

            async with self.session.get(url, params=params) as response:
                if response.status == 200:
                    data = await response.json()

                    # Process forecast data
                    forecasts = []
                    for item in data["list"][:days]:  # Limit to requested days
                        forecasts.append({
                            "date": item["dt_txt"],
                            "temperature_c": item["main"]["temp"],
                            "temperature_f": (item["main"]["temp"] * 9/5) + 32,
                            "description": item["weather"][0]["description"]
                        })

                    return forecasts
                else:
                    return [{"error": f"Forecast API error: {response.status}"}]

        except Exception as e:
            return [{"error": f"Failed to fetch forecast: {str(e)}"}]

    async def cleanup(self):
        """Clean up resources"""
        if self.session:
            await self.session.close()

    Data Processing Skill

    import pandas as pd
from io import StringIO

class DataAnalysisSkill(Skill):
    """Analyze and process data"""

    def __init__(self, config: Dict[str, Any] = None):
        super().__init__(
            config=config,
            dependencies=["memory"]  # For caching results
        )

        self.max_rows = config.get("max_rows", 10000)

    def get_prompts(self) -> List[str]:
        return [
            "You can analyze CSV data and generate insights.",
            "Use load_csv to load data from CSV text.",
            "Use analyze_data to get statistical summaries.",
            "Use create_visualization to generate charts.",
            "Always explain your analysis in simple terms."
        ]

    @tool
    def load_csv(self, csv_data: str) -> Dict[str, Any]:
        """Load and validate CSV data"""

        try:
            # Parse CSV
            df = pd.read_csv(StringIO(csv_data))

            # Validate size
            if len(df) > self.max_rows:
                return {
                    "error": f"Dataset too large. Maximum {self.max_rows} rows allowed.",
                    "actual_rows": len(df)
                }

            # Store in context for other tools
            context = self.get_context()
            context["loaded_data"] = df

            return {
                "success": True,
                "rows": len(df),
                "columns": list(df.columns),
                "dtypes": {col: str(dtype) for col, dtype in df.dtypes.items()},
                "preview": df.head().to_dict()
            }

        except Exception as e:
            return {"error": f"Failed to load CSV: {str(e)}"}

    @tool
    def analyze_data(self, columns: List[str] = None) -> Dict[str, Any]:
        """Generate statistical analysis of data"""

        context = self.get_context()
        df = context.get("loaded_data")

        if df is None:
            return {"error": "No data loaded. Use load_csv first."}

        try:
            # Select columns
            if columns:
                df = df[columns]

            # Generate statistics
            numeric_cols = df.select_dtypes(include=['number']).columns
            categorical_cols = df.select_dtypes(include=['object']).columns

            analysis = {
                "summary": {
                    "total_rows": len(df),
                    "numeric_columns": len(numeric_cols),
                    "categorical_columns": len(categorical_cols)
                }
            }

            # Numeric analysis
            if len(numeric_cols) > 0:
                analysis["numeric_stats"] = df[numeric_cols].describe().to_dict()

            # Categorical analysis
            if len(categorical_cols) > 0:
                categorical_stats = {}
                for col in categorical_cols:
                    categorical_stats[col] = {
                        "unique_values": df[col].nunique(),
                        "top_values": df[col].value_counts().head().to_dict()
                    }
                analysis["categorical_stats"] = categorical_stats

            # Missing values
            missing = df.isnull().sum()
            analysis["missing_values"] = missing[missing > 0].to_dict()

            return analysis

        except Exception as e:
            return {"error": f"Analysis failed: {str(e)}"}

    @tool
    def create_visualization(self, chart_type: str, x_column: str, y_column: str = None) -> Dict[str, Any]:
        """Create data visualization"""

        context = self.get_context()
        df = context.get("loaded_data")

        if df is None:
            return {"error": "No data loaded. Use load_csv first."}

        try:
            import matplotlib.pyplot as plt
            import base64
            from io import BytesIO

            plt.figure(figsize=(10, 6))

            if chart_type == "histogram" and x_column in df.columns:
                plt.hist(df[x_column], bins=20)
                plt.xlabel(x_column)
                plt.ylabel("Frequency")
                plt.title(f"Histogram of {x_column}")

            elif chart_type == "scatter" and x_column in df.columns and y_column in df.columns:
                plt.scatter(df[x_column], df[y_column], alpha=0.6)
                plt.xlabel(x_column)
                plt.ylabel(y_column)
                plt.title(f"{x_column} vs {y_column}")

            elif chart_type == "bar" and x_column in df.columns:
                value_counts = df[x_column].value_counts().head(10)
                plt.bar(range(len(value_counts)), value_counts.values)
                plt.xticks(range(len(value_counts)), value_counts.index, rotation=45)
                plt.ylabel("Count")
                plt.title(f"Top values in {x_column}")

            else:
                return {"error": f"Unsupported chart type or invalid columns"}

            # Save to base64
            buffer = BytesIO()
            plt.savefig(buffer, format='png', bbox_inches='tight')
            buffer.seek(0)

            image_base64 = base64.b64encode(buffer.getvalue()).decode()
            plt.close()

            return {
                "success": True,
                "chart_type": chart_type,
                "image": f"data:image/png;base64,{image_base64}",
                "description": f"{chart_type.title()} chart of {x_column}" + (f" vs {y_column}" if y_column else "")
            }

        except Exception as e:
            return {"error": f"Visualization failed: {str(e)}"}

    Workflow Orchestration Skill

    class WorkflowSkill(Skill):
    """Orchestrate multi-step workflows"""

    def __init__(self, config: Dict[str, Any] = None):
        super().__init__(
            config=config,
            dependencies=["nli", "discovery"]
        )

        self.workflows = {}
        self.active_workflows = {}

    def get_prompts(self) -> List[str]:
        return [
            "You can orchestrate complex multi-step workflows.",
            "Use create_workflow to define new workflows.",
            "Use execute_workflow to run workflows.",
            "Use check_workflow_status to monitor progress.",
            "Break complex tasks into manageable steps."
        ]

    @tool
    def create_workflow(self, name: str, steps: List[Dict[str, Any]]) -> Dict[str, Any]:
        """Create a new workflow definition"""

        try:
            # Validate steps
            for i, step in enumerate(steps):
                if "name" not in step:
                    return {"error": f"Step {i} missing 'name' field"}

                if "action" not in step:
                    return {"error": f"Step {i} missing 'action' field"}

            # Store workflow
            workflow = {
                "name": name,
                "steps": steps,
                "created": time.time()
            }

            self.workflows[name] = workflow

            return {
                "success": True,
                "workflow": name,
                "steps": len(steps),
                "description": f"Created workflow '{name}' with {len(steps)} steps"
            }

        except Exception as e:
            return {"error": f"Failed to create workflow: {str(e)}"}

    @tool
    async def execute_workflow(self, workflow_name: str, inputs: Dict[str, Any] = None) -> Dict[str, Any]:
        """Execute a workflow"""

        if workflow_name not in self.workflows:
            return {"error": f"Workflow '{workflow_name}' not found"}

        workflow = self.workflows[workflow_name]
        execution_id = f"{workflow_name}_{int(time.time())}"

        # Initialize execution state
        execution = {
            "id": execution_id,
            "workflow": workflow_name,
            "status": "running",
            "current_step": 0,
            "results": [],
            "start_time": time.time(),
            "inputs": inputs or {}
        }

        self.active_workflows[execution_id] = execution

        try:
            # Execute each step
            for i, step in enumerate(workflow["steps"]):
                execution["current_step"] = i

                # Execute step based on action type
                if step["action"] == "agent_query":
                    result = await self._execute_agent_step(step, execution)
                elif step["action"] == "data_processing":
                    result = await self._execute_data_step(step, execution)
                elif step["action"] == "wait":
                    result = await self._execute_wait_step(step, execution)
                else:
                    result = {"error": f"Unknown action: {step['action']}"}

                execution["results"].append({
                    "step": step["name"],
                    "action": step["action"],
                    "result": result,
                    "timestamp": time.time()
                })

                # Check for failure
                if result.get("error"):
                    execution["status"] = "failed"
                    execution["error"] = result["error"]
                    break

            else:
                execution["status"] = "completed"

            execution["end_time"] = time.time()
            execution["duration"] = execution["end_time"] - execution["start_time"]

            return {
                "execution_id": execution_id,
                "status": execution["status"],
                "steps_completed": len(execution["results"]),
                "duration": execution["duration"],
                "results": execution["results"][-3:]  # Last 3 results
            }

        except Exception as e:
            execution["status"] = "error"
            execution["error"] = str(e)

            return {
                "error": f"Workflow execution failed: {str(e)}",
                "execution_id": execution_id
            }

    async def _execute_agent_step(self, step: Dict, execution: Dict) -> Dict:
        """Execute agent query step"""

        nli = self.agent.skills.get("nli")
        if not nli:
            return {"error": "NLI skill not available"}

        agent_name = step.get("agent", "general-assistant")
        query = step.get("query", "")

        # Substitute variables
        query = self._substitute_variables(query, execution)

        result = await nli.query_agent(agent_name, query)

        if result.get("success"):
            return {"response": result["response"]}
        else:
            return {"error": result.get("error", "Agent query failed")}

    async def _execute_data_step(self, step: Dict, execution: Dict) -> Dict:
        """Execute data processing step"""

        # Simple data processing example
        operation = step.get("operation", "pass")
        data = step.get("data", execution.get("inputs", {}))

        if operation == "transform":
            # Apply transformation
            transform = step.get("transform", {})
            result = self._apply_transform(data, transform)
            return {"data": result}

        elif operation == "filter":
            # Apply filter
            filter_rule = step.get("filter", {})
            result = self._apply_filter(data, filter_rule)
            return {"data": result}

        else:
            return {"data": data}

    async def _execute_wait_step(self, step: Dict, execution: Dict) -> Dict:
        """Execute wait step"""

        import asyncio

        seconds = step.get("seconds", 1)
        await asyncio.sleep(seconds)

        return {"waited": seconds}

    @tool
    def check_workflow_status(self, execution_id: str) -> Dict[str, Any]:
        """Check status of running workflow"""

        if execution_id not in self.active_workflows:
            return {"error": f"Execution {execution_id} not found"}

        execution = self.active_workflows[execution_id]

        return {
            "execution_id": execution_id,
            "workflow": execution["workflow"],
            "status": execution["status"],
            "current_step": execution["current_step"],
            "steps_completed": len(execution["results"]),
            "duration": time.time() - execution["start_time"]
        }

    import pytest
from unittest.mock import AsyncMock, Mock
from my_custom_skill import WeatherSkill

class TestWeatherSkill:
    @pytest.fixture
    def skill(self):
        return WeatherSkill({
            "api_key": "test_key",
            "base_url": "https://api.test.com"
        })

    @pytest.fixture
    def agent(self, skill):
        from robutler.agents import BaseAgent
        return BaseAgent(
            name="test-agent",
            model="openai/gpt-4o",
            skills={"weather": skill}
        )

    @pytest.mark.asyncio
    async def test_get_weather_success(self, skill):
        """Test successful weather query"""

        # Mock the HTTP session
        mock_response = Mock()
        mock_response.status = 200
        mock_response.json = AsyncMock(return_value={
            "name": "London",
            "main": {"temp": 20, "humidity": 65, "pressure": 1013},
            "weather": [{"description": "sunny"}]
        })

        skill.session = Mock()
        skill.session.get = AsyncMock(return_value=mock_response)

        # Test the tool
        result = await skill.get_weather("London")

        assert result["location"] == "London"
        assert result["temperature_c"] == 20
        assert result["description"] == "sunny"

    @pytest.mark.asyncio
    async def test_get_weather_api_error(self, skill):
        """Test API error handling"""

        mock_response = Mock()

        result = await skill.get_weather("InvalidCity")

        assert "error" in result

    @pytest.mark.asyncio

    Best Practices

    1. Single Responsibility - Each skill should have one clear purpose
    2. Clear Interface - Provide good prompts and tool descriptions
    3. Error Handling - Handle failures gracefully and return meaningful errors
    4. Resource Management - Clean up connections and resources properly
    5. Testing - Write comprehensive tests for all functionality
    6. Documentation - Provide clear examples and API documentation
    7. Configuration - Make skills configurable for different use cases
    8. Dependencies - Declare dependencies explicitly and keep them minimal
